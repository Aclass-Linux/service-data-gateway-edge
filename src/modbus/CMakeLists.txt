# ---- 系统特性检测（替代 Autotools 的 AC_CHECK_*） ----
include(CheckIncludeFile)
include(CheckSymbolExists)
include(CheckStructHasMember)

# 让检测环境与实际编译一致
set(CMAKE_REQUIRED_DEFINITIONS "-D_DEFAULT_SOURCE")

# 函数可用性
check_symbol_exists(strlcpy             "string.h"       HAVE_STRLCPY)
check_symbol_exists(gai_strerror        "netdb.h"        HAVE_GAI_STRERROR)
check_symbol_exists(accept4             "sys/socket.h"   HAVE_ACCEPT4)

# 结构体成员
check_struct_has_member("struct termios2" c_ispeed
  "linux/serial.h" HAVE_STRUCT_TERMIOS2)

# 符号声明
check_symbol_exists(TIOCSRS485  "linux/serial.h" HAVE_DECL_TIOCSRS485)
check_symbol_exists(TIOCM_RTS   "linux/serial.h" HAVE_DECL_TIOCM_RTS)

# 系统头文件
check_include_file("netinet/in.h"  HAVE_NETINET_IN_H)
check_include_file("netinet/ip.h"  HAVE_NETINET_IP_H)

# ---- 生成 config.h（无 .in 模板，直接用 file(WRITE)） ----
set(CONFIG_H
"/* Generated by CMake — system feature detection */
#ifndef CONFIG_H
#define CONFIG_H
")

if(HAVE_STRLCPY)
  string(APPEND CONFIG_H "#define HAVE_STRLCPY\n")
endif()
if(HAVE_STRUCT_TERMIOS2)
  string(APPEND CONFIG_H "#define HAVE_STRUCT_TERMIOS2\n")
endif()
if(HAVE_GAI_STRERROR)
  string(APPEND CONFIG_H "#define HAVE_GAI_STRERROR\n")
endif()
if(HAVE_ACCEPT4)
  string(APPEND CONFIG_H "#define HAVE_ACCEPT4\n")
endif()
if(HAVE_NETINET_IN_H)
  string(APPEND CONFIG_H "#define HAVE_NETINET_IN_H\n")
endif()
if(HAVE_NETINET_IP_H)
  string(APPEND CONFIG_H "#define HAVE_NETINET_IP_H\n")
endif()

# HAVE_DECL_* 需要 0/1（代码用 #if 而非 #ifdef）
if(HAVE_DECL_TIOCSRS485)
  string(APPEND CONFIG_H "#define HAVE_DECL_TIOCSRS485 1\n")
else()
  string(APPEND CONFIG_H "#define HAVE_DECL_TIOCSRS485 0\n")
endif()
if(HAVE_DECL_TIOCM_RTS)
  string(APPEND CONFIG_H "#define HAVE_DECL_TIOCM_RTS 1\n")
else()
  string(APPEND CONFIG_H "#define HAVE_DECL_TIOCM_RTS 0\n")
endif()

string(APPEND CONFIG_H "\n#endif /* CONFIG_H */\n")

file(WRITE ${CMAKE_CURRENT_SOURCE_DIR}/config.h "${CONFIG_H}")

# ---- Modbus 协议库 ----
add_library(modbus
  modbus.c
  modbus-data.c
  modbus-rtu.c
  modbus-tcp.c
)

target_include_directories(modbus
  PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/include
  PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}             # 生成 config.h 所在目录
)

target_compile_definitions(modbus PRIVATE
  _DEFAULT_SOURCE                            # glibc 扩展可见性（需在系统头之前）
)

install(TARGETS modbus
  LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
)

install(FILES
  include/modbus.h
  include/modbus-version.h
  include/modbus-rtu.h
  include/modbus-tcp.h
  DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
)
